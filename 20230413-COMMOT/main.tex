\section{Motivation}

\begin{frame}{Cell-Cell Communication (CCC)}  
    \begin{itemize}
        \item biochemical signaling through \textbf{ligand-receptor (L-R) binding}
        \item inducing \textbf{downstream responses} that shape development, structure, function, etc.
    \end{itemize}
        
    \begin{figure}
        \centering
        \includegraphics[width=0.9\textwidth]{20230413-COMMOT/figs/cellcellcomm.pdf}
        \caption{Approaches and downstream effects of CCC}
        \label{fig:cellcellcomm}
    \end{figure}
\end{frame}

\note[enumerate]{
    \item Just like the society of human beings, 
        the complex structures and functions of multi-cellular creatures are achieved 
        through the coordinated activities of various cells.
    \item Cells make decisions and accomplish their goals by interacting with an environment and other cells.
    \item \textbf{cell-cell communication}, abbreviated as CCC is a major form of such interaction among cells.
    \item Cell-cell communication mainly mediated by the signaling through \textbf{ligand-receptor (L-R) binding}
    \item including autocrine to itself, juxtacrine to its next door, paracrine to its neighbors, and endocrine to some distant cells
    \item and such signals usually induce downstream responses like these.
}







\begin{frame}{Challenges}
    \begin{itemize}
        \item Single-cell based methods obtain \textbf{false positives}
        only taking place \textbf{within limited distance}.
        \item Current spatial data-based methods exam CCC
        only between certain cells \textbf{locally} and
        \textbf{independently} for each L-R pair or cells pair.
    \end{itemize}
    \begin{center}
        \large
        \boxed{\makecell{\texttt{Gitto}, \texttt{CellPhoneDB}, \texttt{stLearn}, \\
        \texttt{SVCA}, \texttt{MISTy}, \texttt{NCEM}, etc.}}
    \end{center}
    \begin{figure}
        \centering
        \includegraphics[width=0.6\textwidth]{20230413-COMMOT/figs/compdist.pdf}
        \caption{Ignored \textbf{\normalsize Competitions} and \textbf{\normalsize distance constraints} of certain L-R bindings}
        \label{fig:compdist}
    \end{figure}
\end{frame}

\note[enumerate]{
    \item Current single-cell based methods have no spatial information of then expression,
        They will produce significant false positives
        that only take place \textbf{within limited spatial distance}
    \item So far, only a few methods were built on spatial data, auch as these
    \item They independently exam CCC on the information between cells or in the neighborhoods of individual cells, 
    so collective or global information in CCC, 
    such as \textbf{competition between cells and molecular species} and \textbf{different distance constraint}, 
    is neglected. 
}







\begin{frame}{Highlights}

    \begin{enumerate}

    \end{enumerate}
    \bigskip

\section{Model}

\subsection{Introduction to Optimal Transport}

\begin{frame}{Optimal Transport
--- Classical: a 2D example}
    Given a transportation planning problem of finding shipment flows $P_{i,j}$ with the least overall cost:
    \begin{enumerate}
        \item $m$ sources with $s_i$ units of \textcolor{blue}{supply} at location $\bm{x}_i$
        \item $n$ targets with $d_j$ units of \textcolor{red}{demand} at location $\bm{y}_j$
        \item shipping $P_{i,j}$ units from $x_i$ to $y_j$ and 
        \item a unit shipment from $\bm{x}_i$ to $\bm{y}_j$ costs $C(\bm{x}_i,\bm{y}_j)$ (such as Euclidean distance)
    \end{enumerate}
    {\small
    \begin{align}
        \bm{P}^*=\argmin_{\bm{P}\in\mathbb{R}_+^{m\times n}} & ~ \langle\bm{P},\bm{C}\rangle_\mathsf{F}
        ~\text{Frobenius inner product}\label{eq:cls}\\ % \langle\bm{P},\bm{C}\rangle_\mathsf{F} \\
        ~~\text{s.t.} & ~ \bm{P1}_n = \bm{s},~ \bm{P}^\mathsf{T}\bm{1}_m = \bm{d},
    \end{align}
    }
    \begin{figure}
        \centering
        \includegraphics[width=0.45\textwidth]{20230413-COMMOT/figs/optrans_demo.pdf}
        \caption{A demo of 2D Optimal Transport}
        \label{fig:optrans_demo}
    \end{figure}
\end{frame}


\note[enumerate]{
    \item Assume that 
        there are 
    \begin{itemize}
        \item $m$ sources with $s_i$ units of {\color{blue}supply} at location $x_i$ and
        \item $n$ targets with $d_j$ units of {\color{red}demand} at location $y_j$
        \item shipping $P_{i,j}$ units from $x_i$ to $y_j$,
            %($\bm{Y}=\{y_j\}_{j=1}^n$ with distribution $d_j$), 
        \item and a unit shipment from $x_i$ to $y_j$ costs $C(x_i,y_j)$, such as it can be Euclidean distance.
    \end{itemize}
    \item \textbf{Optimal transport plan} is a shipment flow with the least overall cost
    \item \uline{the {\color{red}demands} are satisfied by the {\color{blue}supplies}} and 
    \item This a linear programming problem.
    \item where $\bm{P}=[P_{i,j}]_{m\times n}$, $\bm{C}=[C(\bm{x}_i,\bm{y}_j)]_{m\times n}$, 
    $\bm{s}=[s_i]_m$, $\bm{d}=[d_j]_n$, and 
    \item $\langle\bm{P},\bm{C}\rangle_\mathsf{F}=\sum_{i,j}P_{i,j}C(\bm{x}_i,\bm{y}_j)$ is Frobenius inner product.
}



\begin{frame}{Optimal Transport
--- Entropy regularization}
    \textbf{Entropic regularized optimal transport} ($H(\bm{P})=\sum_{i,j}P_{i,j}(\ln P_{i,j}-1)$):
    \begin{align}
        \bm{P}^*=\argmin_{\bm{P}} & ~ \langle\bm{P},\bm{C}\rangle_\mathsf{F} + \epsilon H(\bm{P})\\
        ~~\text{s.t.} & ~ \bm{P1}_n = \bm{s},~ \bm{P}^\mathsf{T}\bm{1}_m = \bm{d}.
    \end{align}
    
    \begin{itemize}
        \item reduce computational complexity
        \item enforce the non-negativity constraint $P_{ij}\in\mathbb{R}_+$
        \item enable efficient numerical solutions
    \end{itemize}

    \begin{figure}
        \centering
        \begin{columns}
            \column{0.55\linewidth}
            \includegraphics[width=0.95\textwidth,right]{20230413-COMMOT/figs/entropyreg.pdf}
            \column{0.3\linewidth}
            \caption{Effect of $\epsilon$ on OT, a example of two 1D distributions}
            {\footnotesize
            The smaller $\epsilon$ is, the more sparse the solution is.
            
            }
        \end{columns}
        \label{fig:entropyreg}
    \end{figure}
    % }
\end{frame}

\note[enumerate]{
    \item A term of entropy regularization $H(\bm{P})=\sum_{i,j}P_{i,j}(\ln P_{i,j}-1)$
    is added to the objective (\ref{eq:cls}) to get \textbf{entropic regularized optimal transport}:
    \begin{itemize}
        \item It is widely used to reduce computational complexity;%\footnote{ ($O(d^3\log d)\to \sim O(d)$), $d$ is the bins of discretization};
        \item Its entropy barriers enforce the non-negativity constraint $P_{ij}\in\mathbb{R}_+$;
        \item It enforces a strong convexity, enabling efficient numerical solutions.
    \end{itemize}
    \item Let's see the effect of $\epsilon$: the smaller $\epsilon$ is, the more sparse the solution is.
}






\begin{frame}{Optimal Transport
--- Unbalanced}
    \textbf{Unbalanced optimal transport} ($\sum_i s_i \neq \sum_j d_j$):
    \begin{itemize}
        \item penalization on the difference of {\color{teal}transported mass} and {\color{violet}anticipated mass}
    \end{itemize}
    
    \begin{align}
        \bm{P}^* = \argmin_{\bm{P}\in\mathbb{R}^{m\times n}} & ~ \langle\bm{P},\bm{C}\rangle_\mathsf{F} 
        + \epsilon H(\bm{P})
        + \rho[
        F({\color{teal}\bm{P1}_n},{\color{violet}\bm{s}})
        +F({\color{teal}\bm{P}^\mathsf{T}\bm{1}_m},{\color{violet}\bm{d}})],
    \end{align}
    where $F(\cdot,\cdot)$ is the penalty measuring the marginal deviation, 
    e.g.
    \begin{itemize}
        \item (in common form) KL divergence: $F(\bm{a},\bm{b})=\infdiv{\bm{a}}{\bm{b}}$
        \item $\ell_1$ penalty: $F(\bm{a},\bm{b})=\|\bm{b}-\bm{a}\|_1$
        \item $\ell_2$ penalty: $F(\bm{a},\bm{b})=\frac{1}{2}\|\bm{b}-\bm{a}\|_2^2$
    \end{itemize}
\end{frame}


\note[enumerate]{
    \item The two total amounts of supply and demand may not be matched,
    some penalization can be assigned to the different between the {\color{teal}transported mass} and {\color{violet}anticipated mass} to get \textbf{unbalanced optimal transport}.
    \item $F(\cdot,\cdot)$ is the penalty function measuring the marginal deviation
    \item it can be KL divergence, $\ell_1$, or $\ell_2$ penalty.
}





\subsection{Collective Optimal Transport}

\begin{frame}{Collective Optimal Transport (COT)
--- Objective}
    \begin{itemize}
        \item $N_S$ spots/cells in a given spatial transcriptomics dataset
        \item $N_L$ ligand and $N_R$ receptor species
        \item $\Tilde{\bm{P}}\in \mathbb{R}^{N_L\times N_R\times N_S\times N_S}$ multi-species coupling
    \end{itemize}
    \begin{gather}
        \Tilde{\bm{P}}^* = \argmin_{\Tilde{\bm{P}}\in\bm{\Gamma}} \sum_{(i,j)\in\mathcal{I}}
        % \alpha_{(i,j)}
        \langle\Tilde{\bm{P}}_{i,j,\cdot,\cdot},\Tilde{\bm{C}}_{(i,j)}\rangle_\mathsf{F} +\sum_{i}F(\Tilde{\bm{\mu}}_i) + \sum_{j}F(\Tilde{\bm{\nu}}_j)\label{eq:obj4d}
    \end{gather}
    where
    \begin{gather}
        \bm{\Gamma} = \left\{
        \Tilde{\bm{P}}\in\mathbb{R}_+^{N_L\times N_R\times N_S\times N_S}: 
        \begin{matrix*}[l]
        \Tilde{\bm{P}}_{i,j,\cdot,\cdot}=\bm{0}~\text{for}~(i,j)\notin\mathcal{I},\\
        \Tilde{\mu}_{i,k} = X_{i,k} - \sum_{j,l} \Tilde{P}_{i,j,k,l} \geq 0,\\
        \Tilde{\nu}_{j,l} = X_{j,l} - \sum_{i,k} \Tilde{P}_{i,j,k,l} \geq 0
        \end{matrix*}
        \right\}.\label{eq:cons4d}
    \end{gather}
\end{frame}

\note[enumerate]{
    \item let's come back to the scenario of signaling between ligands and receptors
    \item Given a spatial transcriptomics dataset with 
    \item $N_S$ spots/cells 
    whose coverage of gene includes 
    \item $N_L$ ligand species and $N_R$ receptor species,
    \item the collective optimal transport can determines an optimal multi-species coupling
    $\Tilde{\bm{P}}^*$ with size of $N_L\times N_R\times N_S\times N_S$
    \item $\Tilde{P}_{i,j,k,l}$ means the quantity of ligand $i$ transported from spot $k$ to spot $l$ for receptor $j$ 
}





\begin{frame}{Collective Optimal Transport (COT)
--- Notations}
    \begin{itemize}
        \item ${\Tilde{P}_{i,j,k,l}}$:  \textbf{signaling strength} between \uline{ligand $i$} and \uline{receptor $j$} from \uline{sender cell $k$} to \uline{receiver cell $l$}; \\
        \item ${\mathcal{I}}$: \textbf{index set} for the interactable L-R pairs (user-defined or from aggregated database); \\
        \item ${X_{i,k}}$: \textbf{expression level} of gene $i$ on spot $k$; \\
        % \item ${\alpha_{(i,j)}}$: LR species-specific \textbf{weights} preassigned for each LR pair; \\
        \item ${\Tilde{\bm{C}}_{(i,j)}\in\mathbb{R}^{N_S\times N_S}}$: L-R species-specific \textbf{cost matrix}, whose element $[C_{(i,j)}]_{k,l}=\begin{cases}
            \varphi(D_{k,l}) & \text{for}~D_{k,l} \leq T_{(i,j)} \\
            \infty & \text{otherwise}
        \end{cases}$ (predefined for each L-R pair);
        \begin{itemize}
            \item $\varphi(\cdot)$: scaling function,
            \item $D_{k,l}$: Euclidean distance between spots $k$ and $l$,  
            \item $T_{(i,j)}$: the spatial limit between ligand $i$ and receptor $j$;
        \end{itemize}
        \item $\Tilde{\bm{\mu}}_i=[\Tilde{\mu}_{i,k}]_{N_S}$: \textbf{untransported expression} of ligand $i$;
        \item $\Tilde{\bm{\nu}}_j=[\Tilde{\nu}_{j,l}]_{N_S}$: \textbf{unsatisfied expression} of receptor $j$.
    \end{itemize}
\end{frame}


\note[enumerate]{
    \item see slides
}




\begin{frame}{Collective Optimal Transport (COT)
--- Constraints}
    \begin{enumerate}
        \item Signaling only occurs between interactable L-R pair
        \begin{gather}
            \Tilde{\bm{P}}_{i,j,\cdot,\cdot}=\bm{0}~\text{for}~(i,j)\notin\mathcal{I}
        \end{gather}
        \item A given ligand/receptor species and sender/receiver cell has limited capacity for interactions (competition)\footnote{e.g. a stronger inferred interaction with one ligand species or cell reduces the potential of interaction with other ligand species or cells}
        \begin{gather}
            \begin{cases}
            \Tilde{\mu}_{i,k} \geq 0\\
            \Tilde{\nu}_{j,l} \geq 0
            \end{cases},
        \end{gather}
    \end{enumerate}
\end{frame}


\note[enumerate]{
    \item The first constraint requires that signaling only occurs between interactable L-R pair
    \item The second require that a given ligand/receptor species and sender/receiver cell has limited
capacity for interactions
    \item which means 
    \begin{itemize}
        \item the total signal sent by spot $k$ through ligand $i$ cannot exceed the expression of ligand $i$ on spot $k$, 
        \item and the total signal received by spot $l$ through receptor $j$ cannot exceed the expression of receptor $j$ on spot $l$.
    \end{itemize}       
}



\begin{frame}{Collective Optimal Transport (COT)
--- 2D Matrix Form}

    {\small\begin{gather}
        {\bm{P}} = \begin{bmatrix}
        \Tilde{\bm{P}}_{1,1,\cdot,\cdot} & \cdots & \Tilde{\bm{P}}_{1,N_R,\cdot,\cdot} \\
        \vdots & \ddots & \vdots \\
        \Tilde{\bm{P}}_{N_L,1,\cdot,\cdot} & \cdots & \Tilde{\bm{P}}_{N_L,N_R,\cdot,\cdot} 
        \end{bmatrix}
        % \in\mathbb{R}^{m\times n}
        ,~
        % \Tilde{C}_{(i-1)\times N_S + k,(j-1)\times N_S + l} = C_{i,j,k,l} \\ \implies
        {\bm{C}} = \begin{bmatrix}
            \Tilde{\bm{C}}_{(1,1)} & \cdots & \Tilde{\bm{C}}_{(1,N_R)} \\
            \vdots & \ddots & \vdots \\
            \Tilde{\bm{C}}_{(N_L,1)} & \cdots & \Tilde{\bm{C}}_{(N_L,N_R)} 
        \end{bmatrix}
        \in\mathbb{R}^{m \times n} \\
        \bm{s} = \begin{bmatrix}
            \bm{X}_{L_1,\cdot} & \cdots & \bm{X}_{L_{N_L},\cdot}
        \end{bmatrix}^\mathsf{T} \in \mathbb{R}^{m},~
        \bm{d} = \begin{bmatrix}
            \bm{X}_{R_1,\cdot} & \cdots & \bm{X}_{R_{N_R},\cdot}
        \end{bmatrix}^\mathsf{T} \in \mathbb{R}^{n}
    \end{gather}}
    where $m:=N_L N_S$, $n:=N_R N_S$, and $\Tilde{\bm{C}}_{(i,j)}\triangleq \infty$ if $(i,j)\notin\mathcal{I}$. 
    Then the original optimization target Equations (\ref{eq:obj4d}) and (\ref{eq:cons4d}) can be rewritten as 
    % \begin{framed}
    % \begin{equation}
    \begin{empheq}[box=\widefbox]{align}
        \min_{{\bm{P}},{\bm{\mu}},{\bm{\nu}}} &~ 
        \langle{\bm{P}},{\bm{C}}\rangle_\mathsf{F} + F(\bm{\mu}) + F(\bm{\nu}) \\
        \text{s.t} &~ 
        {\bm{P}}\in\mathbb{R}_+^{m\times n},~
        {\bm{\mu}}\in\mathbb{R}_+^{m},~
        {\bm{\nu}}\in\mathbb{R}_+^{n},\\
        &~ {\bm{P}}\bm{1}_n = \bm{s} - \bm{\mu},~
        {\bm{P}}^\mathsf{T}\bm{1}_m = \bm{d} - \bm{\nu}
    \end{empheq}
    % \end{equation}
    % \end{framed}
\end{frame}

\note[enumerate]{
    \item The above 4D collective transport optimization problem can be written as a special 2D unbalanced one.
    \item the distances for uninteractable L-R pair are imputated as infinity.
    \item the 2D matrices are assembled by placing each matrix for ligand by row and each matrix for receptor by column.
    \item and array the expression of all ligand/receptor for all spots in a single vector.
    \item the size of transport plan matrix, distance constraint matrix, and expression vectors are denoted as m, the product of ligand number and spot number, and n, the product of receptor number and spot number.
    \item Then the following optimization problem is the same with the previous 4D one.
}




\begin{frame}{COT model with Penalities\footnotemark[1]}
    \begin{exampleblock}{COT with $\ell_1$ penality}
    \begin{align}
        \min_{{\bm{P}},{\bm{\mu}},{\bm{\nu}}} &~ 
        \langle{\bm{P}},{\bm{C}}\rangle_\mathsf{F} 
        + \epsilon_p H(\bm{P}) 
        + \epsilon_\mu H(\bm{\mu}) 
        + \epsilon_\nu H(\bm{\nu})
        + \rho(\|\bm{\mu}\|_1 + \|\bm{\nu}\|_1)\\
        \text{s.t} &~ {\bm{P}}\bm{1}_n = \bm{s} - \bm{\mu},~
        {\bm{P}}^\mathsf{T}\bm{1}_m = \bm{d} - \bm{\nu}
    \end{align}
    \end{exampleblock}
    \begin{exampleblock}{COT with $\ell_2$ penality}
    \begin{align}
        \min_{{\bm{P}},{\bm{\mu}},{\bm{\nu}}} &~ 
        \langle{\bm{P}},{\bm{C}}\rangle_\mathsf{F} 
        + \epsilon_p H(\bm{P}) 
        + \epsilon_\mu H(\bm{\mu}) 
        + \epsilon_\nu H(\bm{\nu})
        + \frac{1}{2}\rho(\|\bm{\mu}\|_2^2 + \|\bm{\nu}\|_2^2)\\
        \text{s.t} &~ {\bm{P}}\bm{1}_n = \bm{s} - \bm{\mu},~
        {\bm{P}}^\mathsf{T}\bm{1}_m = \bm{d} - \bm{\nu}
    \end{align}
    \end{exampleblock}

    \footnotetext[1]{solved by applying Lagrange dual problem and the log-sum-exp trick.}
\end{frame}

\note[enumerate]{
    \item In this paper, the penalty for untransported and unsatisfied masses is L1 or L2 loss
    \item All of these can be solved using Lagrange dual problem and the log-sum-exp trick.
}




\subsection{Utilities for Analysis}
\begin{frame}{Utilities
--- Spatial signaling direction}
    \textbf{Vector field of signaling direction}: $\bm{V}\in\mathbb{R}^{N_S\times d}$ given a L-R pair ($i,j$)'s CCC matrix $\bm{S}=\Tilde{\bm{P}}_{i,j,\cdot,\cdot}\in\mathbb{R}_+^{N_S\times N_S}$
    \begin{align}
        \bm{V}_{k}^\text{{\color{blue}s}}
        & = N\left(\sum_{l\in\mathcal{J}^{s}_k}N\left(\bm{x}_l-\bm{x}_k\right)S_{k,l}\right)\sum_l S_{k,l}\\
        \bm{V}_{l}^\text{{\color{red}r}}
        & = N\left(\sum_{k\in\mathcal{J}^{r}_l}N\left(\bm{x}_k-\bm{x}_l\right)S_{k,l}\right)\sum_l S_{k.l}
    \end{align}
    where \begin{itemize}
        \item $\bm{x}_k$ is the position of spot $k$
        \item $N(\bm{x})={\bm{x}}/{\|\bm{x}\|}$
        \item $\mathcal{J}^\text{s}_k$ is the index set of top $K$ signal-receiving spots 
        with the largest signal for the sender spot $k$, vice versa for receiver spots
    \end{itemize}
\end{frame}

\note[enumerate]{
    \item To visualize the spatial signaling directions,
    a spatial vector field was estimated given a CCC matrix $\bm{S}\in\mathbb{R}_+^{N_S\times N_S}$,
    which is a block we just placed in the 2D matrix of $\bm{P}$ one-by-one.
    \item \begin{itemize}
        \item $d$ is the dimension of space, 
        \item $S_{k,l}$ is the signaling strength 
        \item sent from spot $k$ to spot $l$
        \item \textbf{the $k/l$-th row of $\bm{V}$, $\bm{V}_{k/l}\in\mathbb{R}^d$, represents the signaling direction {\color{blue}sent}/{\color{red}received} by spot $k/l$.}
    \end{itemize}
    \item $\bm{x}_k$ is the position of spot $k$
    \item $N(\bm{x})$ is the vector's direction with unit norm, 
    \item $\mathcal{J}^\text{s}_k$ is the index set of top $K$ signal-receiving spots 
    with the largest signal for the sender spot $k$, vice versa for receiver spots
    \item \textbf{Intuitively, this the summary direction weighted by the top $K$ signal}
}




\begin{frame}{Utilities
--- Cluster-level CCC and its Significance}
    \begin{itemize}
        \item \textbf{Cluster-by-cluster matrix} $\bm{S}^\text{cl}\in\mathbb{R}^{N_C\times N_C}$: averaging spot-by-spot CCC matrix $\bm{S}\in\mathbb{R}^{N_S\times N_S}$ within cluster label:
    \begin{gather}
        S_{k',l'}^\text{cl} = \frac{1}{|\mathcal{J}^\text{cl}_{k',l'}|}\sum_{(k,l)\in\mathcal{J}^\text{cl}_{k',l'}} S_{k,l},
    \end{gather}
    where $\mathcal{J}^\text{cl}_{k',l'}=\{(k,l):L_k=k',L_l=l'\}$ 
    and $L_{k/l}$ and $k'/l'$ are the cluster label of spot $k/l$.

    \item \textbf{Significance} ($p$-value) of cluster-level CCC: permutation of all spots or the spots within cluster.
    \end{itemize}
\end{frame}


\note[enumerate]{
    \item To elucidate CCC among cell states or local groups of spots,
    the spot-by-spot CCC matrix is aggregated to a cluster-by-cluster matrix by averaging the signals within clusters.
    \item To compute the significance ($p$-value) of cluster-level CCC,
    multiple permutations of the locations of all spots or the spots within each cluster was conducted.
}



\begin{frame}{Utilities
--- Evaluation metrics}
    \begin{exampleblock}{\textbf{Cosine distance} of direction field $\bm{V}$ for subsampled data}
    \small
        \begin{gather}
            d_\text{cos}\left(\bm{V}^\text{full},\bm{V}^\text{sub}\right) 
            = \frac{1}{\sum_k\left\|\bm{V}^\text{full}_k\right\|}
            \sum_k\left\|\bm{V}^\text{full}_k\right\|
            \left(1 - \frac{\bm{V}^\text{full}_k\cdot\bm{V}^\text{sub}_k}{\left\|\bm{V}^\text{full}_k\right\|\left\|\bm{V}^\text{sub}_k\right\|}\right)
        \end{gather}
    \end{exampleblock}
    \begin{exampleblock}{\textbf{Jaccard distance} of binarized cluster-level CCC $\bar{\bm{S}}^\text{cl}$}
    \small
        \begin{gather}
            d_\text{jaccard}\left(\bar{\bm{S}}^\text{cl}_1,\bar{\bm{S}}^\text{cl}_2\right)
            = 1 - \frac{\bar{\bm{S}}^\text{cl}_1 \cap\bar{\bm{S}}^\text{cl}_2}
            {\bar{\bm{S}}^\text{cl}_1 \cup \bar{\bm{S}}^\text{cl}_1}
        \end{gather}
    \end{exampleblock}
\end{frame}

\note[enumerate]{
    \item The dissimilarity between two direction field is the weighted average of the direction's cosine distance for each spot.
    \item the dissimilarity between two clustering-level cell-cell communication is the jaccard distance of their signaling graphs with significant edges.
}

\subsection{Simulation study}
\begin{frame}{Simulation 
--- Data generating model}
    \textbf{PDE model with essential physical processes of L-R interaction}
    \begin{itemize}
        \item diffusivity of ligands in space ($D_i$),
        \item binding and dissociation of L-R complexes ($a_{(i,j)}$ and $b_{(i,j)}$), and 
        \item degradation of molecules ($c_{(i,j)}$).
    \end{itemize}
    {\small
    \begin{align}
        \frac{\partial {\color{blue}[L_i]}}{\partial t} 
        & = D_i\nabla^2{\color{blue}[L_i]} 
        - \sum_{j:(i,j)\in\mathcal{I}}a_{(i,j)}{\color{blue}[L_i]}{\color{red}[R_j]} 
        + \sum_{j:(i,j)\in\mathcal{I}}b_{(i,j)}{\color{violet}[L_i R_j]}
        - c_i{\color{blue}[L_i]}, \\
        \frac{\partial {\color{violet}[L_i R_j]}}{\partial t}
        & = a_{(i,j)}{\color{blue}[L_i]}{\color{red}[R_j]}
        - b_{(i,j)} {\color{violet}[L_i R_j]}, \\
        \frac{\partial {\color{red}[R_j]}}{\partial t}
        & = - \sum_{i:(i,j)\in\mathcal{I}} a_{(i,j)}{\color{blue}[L_i]}{\color{red}[R_j]}
        + \sum_{i:(i,j)\in\mathcal{I}} b_{(i,j)}{\color{violet}[L_i R_j]},
    \end{align}
    where ${\color{blue}[L_i]}$, ${\color{red}[R_j]}$, and ${\color{violet}[L_i R_j]}$: $\mathbb{R}_+^{d + 1}\to\mathbb{R}_+$ denote the spatial distributions of 
    {\color{blue}ligand $i$}, {\color{red}receptor $j$}, and the {\color{violet}complex} formed by their binding changing with time $t$.
    }
\end{frame}

\note[enumerate]{
    \item CCC in space is simulated by a partial differential equation (PDE) model,
    which describes the essential physical processes in L-R interaction in 2D space, including:
    \begin{itemize}
        \item diffusivity of ligands in space ($D_i$),
        \item binding and dissociation of L-R complexes ($a_{(i,j)}$ and $b_{(i,j)}$), and 
        \item degradation of molecules ($c_{(i,j)}$).
    \end{itemize}
    \item ${\color{blue}[L_i]}$, ${\color{red}[R_j]}$, and ${\color{violet}[L_i R_j]}$: $\mathbb{R}_+^{d + 1}\to\mathbb{R}_+$ denote the spatial distributions of 
    {\color{blue}ligand $i$}, {\color{red}receptor $j$}, and the {\color{violet}complex} formed by their binding changing with time $t$.
    \item The initial distributions of ligand and recepter are randomized normally,
    \item The initial distributions of complex is 0,
    \item and the distribution of complex in equilibrium time depicts the ``true'' L-R binding states.
}

\section{Summary}

\begin{frame}{Summary}
    \begin{enumerate}
        \item \texttt{COMMOT} can infer CCC for all L-R species simultaneously, visualize spatial CCC at various scales including spatial signaling directions, and analyze their downstream effects.
        \item The collective optimal transport backend incorporates both competing marginal distributions 
        and constrained transport plans, which cannot be dealt with by current optimal transport methods.
        % \item This method can be applied on a wide range of data types with different spatial resolutions and gene coverage. 
        \item It can consistently capture the CCC activities known from the literature and 
        discovers new signaling which can be confirmed by immunofluorescence staining.
        \item With higher correlation between inferred L-R activities and their downstream genes, it outperformed current CCC inference methods.
        % higher WNT signaling increases the expression of several genes,
    \end{enumerate}
\end{frame}
