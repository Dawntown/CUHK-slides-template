\section{Motivation}

\begin{frame}{Perturbation in biology}
    \begin{itemize}\small
        \item Perturbation: molecular responses (treatment effect) to external signals
        \begin{itemize}
            \item drug treatment, genetic interference, pathogen infection, etc
        \end{itemize}
        \item Individual treatment effect (ITE): the gene expression \textbf{changes in single cell before and after treatment}
        \begin{gather}
            \text{ITE} = \mathbf{X}_{t=1} - \mathbf{X}_{t=0}.
        \end{gather}
    \end{itemize}
    
    \textbf{Challenges}
    \begin{enumerate}
        \item Perturbed and non-perturbed cells are unpaired since destruction when measuring.
        \item Underlying heterogeneity may obscure the estimation of ITE. 
        \begin{itemize}
            \item cell type, cell cycle, micro-environment, etc
        \end{itemize}
    \end{enumerate}

    \begin{figure}
        \centering
        \includegraphics[width=0.7\linewidth]{20240314-Perturbation/figs/pert.png}
        \caption{Perturbation experiment and response heterogeneity.}
        \label{fig:pert}
    \end{figure}
\end{frame}

\note[enumerate]{
    \item Understanding molecular responses in single cells upon various perturbations is crucial in biology.
    \item Single-cell measurements typically destroy the cells, leading to \textbf{unpaired} distributions of perturbed and non-perturbed cells.
    \item Heterogeneous confounding factors such as the cell statement, and the micro-environment
}{}


\begin{frame}{Causal inference framework}
    Counterfactual describes the treatment effect on untreated cells (vice versa) 
    by constructing causal pair that have matched confounding conditions. 
    Set {$z$ to select a single cell}, then
    \begin{align}
        \text{ITE} & = \mathbb{E}[X_{T=1}\mid T=0, Z=z] - \mathbb{E}[X\mid T=0, Z=z]
    \end{align}
    
    \begin{definition}[Counterfactual]
        \textbf{Counterfactual outcome}: An expression ($X$) this cell should have had if \textit{it was treated} ($T=1$) rather than \textit{doing nothing to it} ($T=0$).
    \end{definition}
    \medskip

    \begin{framed}
    \texttt{CINEMA-OT}
    \begin{enumerate}
        \item identifies and separates \textbf{confounding variations} from perturbation effects, and
        \item constructs \textbf{causal pairs of cells} by optimal transport on confounding space.
    \end{enumerate}  
    \end{framed}
\end{frame}


\section{CINEMA-OT}
\begin{frame}{Assumptions}
    \begin{enumerate}
        \item Independent sources and noise: $\vperp \{\bm{s}_{1:L},\bm{e}_{1:L},\epsilon_{i:N}\}$
        \item Linearity of source signal combinations%\footnote{\color{red}Author claimed that at most only one the signal is Gaussian, why?}
    \end{enumerate}
    \begin{gather}
        \bm{z}_i{\sim}[s_1 + e_1 , \cdots , s_L + e_L , t + e_t]\\
        \bm{x}_i = \bm{z}_i^\mathsf{T}\mathbf{A} + \bm{\epsilon}_i
    \end{gather}
    \begin{itemize}\small
        \item $\bm{s}_{1:L}$: confounding factors
        \item $t$: treatment event $t\in\{0,1\}$
        \item $\bm{e}_{\{1:L,t\}}$: biological variation contributing to the heterogeneity within a factor
        \item $\bm{z}_i\in\mathbb{R}^{L+1}$: \textbf{latent} vector for cell $i$, $i=1,\cdots, N$ from the signals
        \item $\bm{\epsilon}_i\in\mathbb{R}^G$: measurement noise for cell $i$, i.i.d.
        \item $\mathbf{A}\in\mathbb{R}^{(L+1)\times D}$: mixing matrix from $L+1$ sources to $D$ genes
        \item $\bm{x}_i\in\mathbb{R}^{G}$: \textbf{observational} gene count vector of cell $i$
    \end{itemize}
\end{frame}

\note[enumerate]{
    \item Confounding factors and treatment event are independent r.v.s.
    \item Confounding factor, such as cell type $s_l\in\{1,\cdots,K\}$
    \item The expression of each gene can be linearly decomposed as the mixing of noisy confounding signals and treatment-associated signals.
}{}

\begin{frame}{Identification of confounding factors}
    \begin{enumerate}
        \item {Decomposing independence components}
        \begin{enumerate}
            \item the first $L+1$ PCs, $\Tilde{\bm{x}}_i$, can be viewed as
            a \textbf{linear transformation of the original signals}
            \begin{gather}
                \Tilde{\bm{x}}_i\overset{\text{iid}}{\sim}\mathbf{B}[s_1 + e_1, \cdots, s_L + e_L, t + e_t]
            \end{gather}
            \item the identified \textbf{ICs can be unmixed up} to a permutation\footnote{by ICA identifiability theorem}
            %requiring the \color{red}at-most-only-one-Gaussian assumption}
            \begin{gather}
                \mathbf{W}^{\text{ICA}}\hat{\bm{x}}_i\overset{iid}{\sim}[s_1 + e_1 , \cdots , s_L + e_L, t + e_t],~\text{up to a permutation}
            \end{gather}
        \end{enumerate}
        \medskip
        \item {Obtaining confounding space} $S^c$
        \begin{enumerate}
            \item \textbf{testing independency} between treatment event and each IC\footnote{Chatterjee's coefficient-based distribution-free test}
            \item decoupling $S^c:=\hat{\bm{s}}_{i,1:L}$ and $\hat{t}_{i}$
        \end{enumerate}
    \end{enumerate} 
\end{frame}

% \note[enumerate]{
%     \item the confounder terms are identifiable with the ICA tranform up to a permutation
%     \item the contributions of noise terms are not distorted as the relative ratio between $s$, $z$ and $e$ are preserved in the output
% }{}

\begin{frame}{Construction of causal pairs --- \normalsize Optimal transport (Recap)}
    Given two distributions $\mu,\nu\in\mathbb{R}^d$ and cost function $C(x,y)$ (such as $\|x-y\|^2$),
    we would like to find a mapping minimizing the total cost
    {\small\begin{gather}
        V(\mu,\nu) = \min_{T:T(\mu)=\nu}\ \mathbb{E}_{X\sim\mu}C(X,T(X)) \label{eq:primalot}
    \end{gather}}
    \begin{itemize}\small
        \item $V$: distance measure between two distributions 
        \item $T^*$: optimal mapping between two distributions
    \end{itemize}
    \bigskip
    By \textbf{regularization}, it become tractable\footnote{It can be efficiently solved by the Sinkhorn–Knopp algorithm}.
    {\small\begin{align}
        \min_{\gamma\in\Gamma}\ & \mathbb{E}_{(X,Y)\sim\gamma}C(X,Y) + \epsilon {\color{blue}H(\gamma)}  \\
        \text{s.t.}\quad
        & \Gamma\triangleq\{\gamma:\mathbb{E}\mathbbm{1}_{(x,Y)}=\mu(x),\mathbb{E}\mathbbm{1}_{(X,y)}=\nu(y)\} \notag \\
        & {\color{blue}H(\gamma) \triangleq \mathbb{E}_{(X,Y)\sim \gamma}\ln \gamma(X,Y) - 1}\notag
    \end{align}}
    \begin{itemize}\small
        \item $\Gamma(\mu,\nu)$: the collection of all couplings (joint dist.\footnote{
        Discretizing $\mu$ and $\nu$ to $N_1$ and $N_2$ bins, respectively, $\gamma \in \mathbb{R}_+^{N_1\times N_2}$.
        }) $\gamma$ between $\mu$ and $\nu$
        \item $H(\gamma)$: entropy regularization
    \end{itemize}
    \smallskip
\end{frame}

\begin{frame}{Construction of causal pairs --- \normalsize mapping between conditions in space $S^c$}
    To pair between treatment ($\{\bm{x}'_i\overset{\text{iid}}{\sim}\mu'\}_{i=1}^{N_1}$) and 
    control ($\{\bm{x}_j\overset{\text{iid}}{\sim}\mu\}_{j=1}^{N_2}$) groups and 
    providing cost function $C(\bm{x},\bm{y}) \triangleq e^{\frac{\|\bm{x}-\bm{y}\|}{s}}$, \texttt{CINEMA-OT} gives
    \begin{enumerate}
        \item mapping matrix (joint dist.):
        \begin{gather}
            \mathbb{R}_+^{N_1\times N_2}\ni\mathbf{M}:=\gamma^*\gets \text{OT}(S^c(\{\bm{x}'_i\}_{i=1}^{N_1}),S^c(\{\bm{x}_j\}_{j=1}^{N_2}))
        \end{gather}
        \item individual treatment effect\footnote{in matrix form, $\text{ITE}=\mathbf{X}'- \diag{(\mathbf{M1})}^{-1}\mathbf{M}$}:
        \begin{align}
            \text{ITE}(\bm{x}'_i) = & \bm{x}'_i - \hat{\bm{x}}_i \\
            \hat{\bm{x}}_i := & \sum_{j=1}^{N_2}\bm{x}_j\frac{M_{i,j}}{\sum_{j'=1}^{N_2}M_{i,j'}}
        \end{align}
    \end{enumerate}
    \textcolor{blue}{Restore the treated cell before treatment by the combination of untreated cells similar in confounding factors.}
\end{frame}


\begin{frame}{Pairing under differential abundance --- \normalsize Weighted optimal transport}
    The cells may die or proliferate in response to a perturbation.
    \begin{itemize}
        \item change the distribution of cell densities and
        \item {introduce \textbf{dependency} between confounding factors and the treatment event}.
    \end{itemize}

    \item A treatment may change the distribution of cell densities, for example cells may die or proliferate in response to a perturbation,
    \textbf{introducing dependency between underlying confounders and the treatment event}.
}{}

\begin{frame}{Pairing under differential abundance --- \normalsize Reweighting procedure}
    \texttt{CINEMA-OT-W}: \textcolor{blue}{grouping the cells with similar confounder characteristics}
    \begin{enumerate}\small
        \item Given PC embeddings of both groups: $\{\Tilde{\bm{x}}'_i\}_{i=1}^{N_1}$ and $\{\Tilde{\bm{x}}_j\}_{j=1}^{N_2}$
        \item Find the $K$-NNs of each treated cell in untreated group: $\{\mathcal{N}_K(\Tilde{\bm{x}}'_i)\}_{i=1}^{N_1}$
        \item Synthesize the matched untreated proxy for each $\Tilde{\bm{x}}'_i$: $\hat{\Tilde{\bm{x}}}_i = \mean(\{\Tilde{\bm{x}}_j\}_{j\in\mathcal{N}_K(\Tilde{\bm{x}}'_i)})$
        \item Cluster the proxies of the treated and emb.'s of the untreated: $\{\hat{\Tilde{\bm{x}}}_{1:N_1},\Tilde{\bm{x}}_{1:N_2}\}$\footnote{The clustering and above can be omitted if there are predefined meaningful labels.}
        \item Downsample cells in each cluster $c$ to have equal densities: $|\mathcal{S}'_c|=|\mathcal{S}_c|$
        \item Perform \texttt{CINEMA-OT} on the downsampled data $\{\bm{x}'_i\}_{i\in \mathcal{S}'}$ and $\{\bm{x}_j\}_{j\in\mathcal{S}}$
    \end{enumerate}
\end{frame}

\begin{frame}{Overview --- \normalsize Methods and notations}
    \begin{enumerate}
        \item Proxy-based methods (Mixscape\cite{papalexi_characterizing_2021} and Harmony\cite{korsunsky_fast_2019}-Mixscape)
        \item Autoencoder-based methods (CPA\cite{lotfollahi_predicting_2023}, scGen\cite{lotfollahi_scgen_2019}, and ContrastiveVI\cite{weinberger_isolating_2023})
        \item Pure OT-based methods (CellOT\cite{bunne_learning_2023})
    \end{enumerate}
    \medskip
    Common notations
    \begin{itemize}\small
        \item Original expression matrices: $\{\bm{x}'_i\}_{i=1}^{N_1}$ and $\{\bm{x}_j\}_{j=1}^{N_2}$
        \item Predicted expression matrices: $\{\hat{\bm{x}}'_i\}_{i=1}^{N_1}$ and $\{\hat{\bm{x}}_j\}_{j=1}^{N_2}$
        \item PC embeddings: $\{\Tilde{\bm{x}}'_i\}_{i=1}^{N_1}$ and $\{\Tilde{\bm{x}}_j\}_{j=1}^{N_2}$
        \item Other latent embeddings: $\{\bm{z}'_i\}_{i=1}^{N_1}$ and $\{\bm{z}_j\}_{j=1}^{N_2}$
    \end{itemize}
    \medskip
    Common targets
    \begin{enumerate}\small
        \item consider underly covariates in model by \textbf{recovering confounder/shared space}
        \item estimate/predict the \textbf{single-cell level treatment effect} of certain perturbation
    \end{enumerate}

\end{frame}

\begin{frame}{Overview --- Objectives and metrics}
    Simulation setting:
    \begin{enumerate}\small
        \item overall treatment effect,
        \item confounder-specific treatment effect, e.g. cell type
        \item unobservable-factor-driven treatment effect, e.g. drug dose distribution
    \end{enumerate}
    \bigskip
    Evaluation metrics
    \begin{enumerate}\small
        \item \textbf{ASW} (average silhouette width) and \textbf{PCR} (principal components regression score) measure \uline{cell distribution equality after treatment-effect removal}.
        \item \textbf{ITE ARI} (adjuested rand index) compares the \uline{overlap between the clustering on estimated differential response and cell population}, 
        indicating the clustering structure preservation on response.
        \item \textbf{ITE PCR} measures the \uline{independency between ground-truth unrelated confounders and response}, indicating the attribution accuracy.
    \end{enumerate}
\end{frame}

\begin{frame}{Autoencoder-based methods --- \normalsize CPA (assumptions)}
    \textbf{Assumptions} of compositional perturbation autoencoder (CPA)\cite{lotfollahi_predicting_2023}: 
    given a dataset $\mathcal{D}=\{(\bm{x}_i,\bm{d}_i,\bm{c}_i)\}_{i=1}^N$, every sample (omit subscript $i$) is generated from a composed low-dimension representation
    \begin{gather}
        \bm{z} = \bm{z}_0 + \mathbf{V}^\text{t}[f_1(d_1),\cdots,f_M(d_M)] + \sum_{j=1}^K\mathbf{V}^\text{c}_j\bm{c}_j
    \end{gather}
    \begin{itemize}\small
        \item $\bm{d}=[d_1,\cdots,d_M]\geq\bm{0}$: doses of $M$ drugs, $d_j=0$: not treated by drug $j$
        \item $\bm{c}=[\bm{c}_1,\cdots,\bm{c}_K]$: additional discrete covariates such as cell types or species, $\bm{c}_j\in\{0,1\}^{K_j}$ is a one-hot vector.
        \item $\bm{z}_0\in\mathbb{R}^d$: basal latent state, \textbf{independent with perturbations/coveriates}
        \item $f_{j}:\mathbb{R}\to\mathbb{R}$: non-linear scale each $d_j$
        \item $\mathbf{V}^\text{t}\in\mathbb{R}^{d\times M}$: embedding for each possible perturbation represented in $\bm{d}$
        \item $\mathbf{V}^\text{c}_j\in\mathbb{R}^{d\times K_j}$: embedding for the $j$-th discrete covariate
    \end{itemize}
    \bigskip
    \textbf{Target}: learn a ML model, given a triplet $(\bm{x}_i,\bm{d}_i,\bm{c}_i)$ as well as a target perturbation $\bm{d}'$, estimating the \textbf{counterfactual gene expression} $\hat{\bm{x}}_i'$.
    \begin{gather}
         \hat{\bm{x}}'_i=\mathcal{M}((\bm{x}_i,\bm{d}_i,\bm{c}_i),\bm{d}')
    \end{gather}
\end{frame}

\begin{frame}{Autoencoder-based methods --- \normalsize CPA (AE architecture)}
    Parameterization of the \textbf{unknown components}
    \begin{enumerate}\small
        \item encoder: $\hat{\bm{z}}_0 = \hat{\bm{f}}^\text{enc}(\bm{x})$
        \item composition: $\hat{\bm{z}} = \hat{\bm{z}}_0 + \hat{\mathbf{V}}^\text{t}[\hat{f}_1(d_1),\cdots,\hat{f}_M(d_M)] + \sum_{j=1}^K\hat{\mathbf{V}}^\text{c}_j\bm{c}_j \ \text{s.t.}\ \hat{f}_{1:M}(0) = 0$
        \item counterfactual mean and variance: $\hat{\bm{f}}^\text{dec}_\mu(\hat{\bm{z}})$ and $ \hat{\bm{f}}^\text{dec}_{\sigma^2}(\hat{\bm{z}})$
    \end{enumerate}
    \medskip
    Three parts of \textbf{loss function}
    \begin{enumerate}\small
        \item reconstruction loss (Gaussian negative log-likelihood), 
        where a positive constraint is imposed on $\sigma^2$ by $s(\sigma^2):=\log(\exp(\sigma^2+10^{-3})+1)$
        \begin{gather}
            l_i 
            = \log s(\hat{\bm{f}}^\text{dec}_{\sigma^2}(\hat{\bm{z}}_i))/2 
            + \|\hat{\bm{f}}^\text{dec}_\mu(\hat{\bm{z}}_i)-\bm{x}_i\|^2/2s(\hat{\bm{f}}^\text{dec}_{\sigma^2}(\hat{\bm{z}}_i))
        \end{gather}
        \item dependency loss (for perturbation/covariates classifiers)
        \begin{gather}
            l_i^\text{d} 
            = \texttt{CE}(\hat{f}^\text{adv}_d(\hat{\bm{z}}_{i,0},\bm{d}_i))\quad \& \quad
            l_{i,j}^\text{c} 
            = \texttt{CE}(\hat{f}^\text{adv}_{c_j}(\hat{\bm{z}}_{i,0},\bm{c}_{i,j}))\ \forall\ j=1:K
        \end{gather}
    \end{enumerate}
    \medskip
    \textbf{Iteratively updating} parameters
    \begin{enumerate}\small
        \item $(\hat{f}^\text{adv}_d, \hat{f}^\text{adv}_{c_{1:K}})\gets\argmin \sum_{i}l_i^\text{d}+\sum_{i,j}l_{i,j}^\text{c}$
        \item $(\hat{\bm{f}}^\text{enc},\hat{\bm{f}}^\text{dec},\hat{f}_{1:M},\hat{\mathbf{V}}^\text{t},\hat{\mathbf{V}}^\text{c}_{1:K})\gets\argmin \sum_{i}l_i-\lambda(\sum_{i}l_i^\text{d}+\sum_{i,j}l_{i,j}^\text{c})$
    \end{enumerate}
\end{frame}

\begin{frame}{Autoencoder-based methods --- \normalsize scGen (assumptions and prediction)}
    \textbf{Assumption} of scGen\cite{lotfollahi_scgen_2019}
    \begin{enumerate}
        \item the high-dimension expression $\bm{X}$ is generated from low-dimension representation $\bm{Z}$ in conditions $\bm{P}$, and
        \item the variability related to conditions (perturbation and other covariates) linearly contributes to the representation.
    \end{enumerate}
    \begin{align}
        \bm{X} & \sim P(\bm{x}\mid \bm{Z}, \bm{P}) \label{eq:scgenvae}\\
        \bm{z} & = \bm{z}_0 + \bm{p}^\mathsf{T}\mathbf{D}
    \end{align}
    E.g. in perturbation, the representation of treated A-typed cell is 
    \begin{gather}
        \bm{z}_{A,t=1} = \bm{z}_{A,t=0} + \bm{\delta}\\
        \implies\hat{\bm{\delta}} = \mean(\hat{\bm{z}}_{A,t=1})-\mean(\hat{\bm{z}}_{A,t=0})
    \end{gather}.
    \begin{enumerate}
        \item Upsample within cell type to reach equal population for each condition
        \item Downsample within condition to reach equal population
        \item scGen constructs VAE to infer $\hat{\bm{z}}_i$ for each cell
    \end{enumerate}
\end{frame}

\begin{frame}{Pure OT-based methods --- \normalsize OT dual problem (OT theory cont.)}
    The primal formulation of OT can also be solved by considering its \textbf{dual problem} if Euclidean distance acts as the cost function.
    {\small
    \begin{align}
        W(\mu,\nu)  & = \min_{\gamma\in\Gamma(\mu,\nu)} \ \mathbb{E}_{(\bm{X},\bm{Y})\sim\gamma}\|\bm{X} - \bm{Y}\|_2^2 \quad\quad\quad\quad \text{(Primal)} \\
        % &&\text{s.t.}\quad & \Gamma\triangleq\{\gamma:\mathbb{E}\mathbbm{1}_{(x,Y)}=\mu(x),\mathbb{E}\mathbbm{1}_{(X,y)}=\nu(y)\} \\
        W(\mu,\nu) & = \max_{(g,f)\in\Phi} \ {\color{teal}\mathbb{E}g(\bm{X})} + \mathbb{E}f(\bm{Y}) \quad\quad\quad\quad\quad\quad \text{(Dual)} \\
        \text{s.t.}\ \Phi & \triangleq\{{(g,f)\in\mathbb{R}\times\mathbb{R}:g(\bm{x}) + f(\bm{y})\leq\frac{1}{2}\|\bm{x}-\bm{y}\|_2^2,a.e.}\}\notag
    \end{align}}
    
    \bigskip
    
    The dual problem can be \textbf{further simplified}\cite{villani2003topics} to 
    {\small
    \begin{align}
        W(\mu,\nu) & = \underbrace{\frac{1}{2}\mathbb{E}[\|\bm{X}\|_2^2 + \|\bm{Y}\|_2^2]}_{=:\mathcal{E}_{\mu,\nu}} - 
        \min_{f\in\Tilde{\Phi}}\{\mathbb{E} {\color{teal}f^*(\bm{X})} + \mathbb{E} f(\bm{Y})\}\label{eq:dualot}\\
        \text{s.t.}\ \Tilde{\Phi} & \triangleq \{{f\in\mathbb{R}:\text{convex functions}}\}\notag
    \end{align}}
    \begin{itemize}
        \item ${\color{teal}f^*(\bm{x}) \triangleq \max_{\bm{y}}\ \left[\langle\bm{x},\bm{y}\rangle - f(\bm{y})\right]}$: $f$'s convex conjugate
        \begin{itemize}
            \item optimal transport map $T=\nabla f^*$ satisfying $\bm{X}\sim\mu\implies T(\bm{X})\sim\nu$
            \item convex conjugate $f^*$ is \textbf{very hard to compute}.
        \end{itemize}
    \end{itemize}
\end{frame}


\begin{frame}{Pure OT-based methods --- \normalsize Approximation in a convex set}
    Another convex function $g$ is proposed to approximate the simplified dual problem\cite{pmlr_v119_makkuva20a}, Eq. (\ref{eq:dualot}). 
    Since ${\nabla f^*(\bm{x})\triangleq \argmax_{\bm{y}}\ \left[\langle\bm{x},\bm{y}\rangle - f(\bm{y})\right]}$, observe that for all $g\in\Tilde{\Phi}$
    \begin{gather}
         \langle \bm{x}, \nabla g(\bm{x})\rangle - f(\nabla g(\bm{x})) 
         \leq \langle \bm{x}, \nabla f^*(\bm{x})\rangle - f(\nabla f^*(\bm{x})) 
         = f^*(\bm{x}) \\
         \implies \mathbb{E}\left[\langle\bm{X},\nabla g(\bm{X})\rangle - f(\nabla g(\bm{X}))\right]
         \leq \mathbb{E} f^*(\bm{X}),
    \end{gather}
     where the equality is achieved with $g=f^*$, i.e.
     \begin{gather}
         \max_{g\in\Tilde{\Phi}}\ \mathbb{E}\left[\langle\bm{X},\nabla g(\bm{X})\rangle - f(\nabla g(\bm{X}))\right]
         = \mathbb{E} f^*(\bm{X})
     \end{gather}
     Then Eq. (\ref{eq:dualot}) can be approximated as
     \begin{gather}
     \small
         W(\mu,\nu) = \mathcal{E}_{\mu,\nu} - 
         \min_{f\in\Tilde{\Phi}} \max_{g\in\Tilde{\Phi}}\underbrace{\left\{
            \mathbb{E}\left[\langle\bm{X},\nabla g(\bm{X})\rangle - f(\nabla g(\bm{X}))\right] + 
            \mathbb{E}f(\bm{Y})
        \right\}}_{=:\mathcal{V}_{\mu,\nu}(g,f)}\label{eq:convexot}
     \end{gather}
\end{frame}


\begin{frame}{Pure OT-based methods --- \normalsize Input-convex neural networks (ICNNs)}
    Convex spaces, such as $\Tilde{\Phi}$ in Eq. (\ref{eq:dualot}), can be parameterized with neural networks convex w.r.t. to the inputs 
    by setting certain constraints on the parameters of fully connected FFN, $\bm{\theta}=\{\bm{b}_l,\mathbf{W}_l\}_{l=1}^L$.
    \begin{gather}
        \bm{h}_{l} \gets \sigma_l(\mathbf{W}_l\bm{h}_{l-1} + \bm{b}_l),l=1,\cdots,L;\quad \bm{h}_L \gets f(\bm{x};\bm{\theta})
    \end{gather}
    \begin{itemize}\small
        \item $\sigma_l$: convex and non-decreasing activation functions
        \item $\mathbf{W}_l$: therein all elements are constrained to be nonnegative
    \end{itemize}
    \bigskip
    \begin{itemize}
        \item Any convex function over a convex domain can be approximated in sup norm by an ICNN.
        % \item ICNNs have been utilized to parameterize Wasserstein gradient flows as well as barycenters.
    \end{itemize}
\end{frame}

\begin{frame}{Pure OT-based methods --- \normalsize Parameterizing OT by ICNNs}
    The convex functions in Eq. (\ref{eq:convexot}) can be parameterized by ICNNs, 
    i.e. $g_{\bm{\theta}}$ and $f_{\bm{\phi}}$
    \begin{align}
        (g_{\bm{\theta}^*},f_{\bm{\phi}^*}) & \gets \arg\max_{\bm{\phi}}\min_{\bm{\theta}}  \{\- \mathcal{V}_{\mu,\mu'}(g_{\bm{\theta}},f_{\bm{\phi}})
        \} \\
        & \quad\quad\quad\text{s.t.} \mathbf{W}_l \geq 0,\ \forall \mathbf{W}_l \in \bm{\theta}\cup\bm{\phi}\, \forall l \\
        T^* & = \nabla g_{\bm{\theta}^*} 
        \implies \hat{\bm{x}}_j' = \nabla g_{\bm{\theta}^*}(\bm{x}_j)
    \end{align}
    \begin{itemize}\small
        \item $\bm{X}\sim\mu$: control group; $\bm{X}'\sim\mu'$: treatment group 
        \item $\bm{\theta},\bm{\phi}$: the set of parameters in ICNNs
        \item $T^*$: the optimal mapping such that $\mu' = (T^*)_\#\mu$
    \end{itemize}
    \medskip
    \color{gray}{\small
    In practice, the negative weights constraints of $\bm{\theta}$ usually be relaxed by introducing a penalty on it.
    \begin{align}
        \max_{\bm{\phi}:\mathbf{W}_l\geq 0,\forall l}
        \min_{\bm{\theta}}\ &
        \left\{
        \mathbb{E}\left[f_{\bm{\phi}}(\nabla g_{\bm{\theta}}(\bm{X})) - \langle\bm{X},\nabla g_{\bm{\theta}}(\bm{X})\rangle\right] - 
            \mathbb{E}f_{\bm{\phi}}(\bm{X}') + \lambda R(\bm{\theta})
        \right\} \\
        \text{s.t.}\ & R(\bm{\theta})\triangleq \sum_{\mathbf{W}_l\in\bm{\theta}}\|\max(-\mathbf{W}_l,0)\|_F^2
    \end{align}
    }
\end{frame}

\begin{frame}{Discussion}
    Highlight:
    \begin{enumerate}
        \item \texttt{CINEMA-OT} produces a per-cell view into the effects of treatment and conditions by
        \begin{enumerate}
            \item explicitly separating confounder and treatment signals and
            \item matching the cells at the single-cell level to form causal pairs with similar confounders.
        \end{enumerate}

        \item \texttt{CINEMA-OT} outperforms other methods in synthetic and real datasets.
    \end{enumerate}

    \bigskip
    Challenges
    \begin{enumerate}
        \item bias-variance trade-offs in optimal transport
        \item batch effect hidden in estimated biological perturbation effect
        \item disability to predict the treatment effect of unmeasured perturbation-cell pairs
        \item disability to model the changes in confounder distribution brought by treatment
    \end{enumerate}
\end{frame}
