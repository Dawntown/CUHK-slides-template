\section{Introduction}

\begin{frame}{Antibody \normalsize --- A brief introduction}
    \begin{itemize}\small
        \item \textbf{CDR ({Complementarity-determining region})}: small and highly variable regions, accounting for the antibody's specificity;
        \item \textbf{Framework region}: the rest parts in variable regions;
        \item \textbf{Fab ({Fragment antigen-binding})}: part of antibody binding to antigens;
        \item \textbf{Fc ({Fragment crystallizable})}: tail part of antibody interacting with cell surface receptors or complement system.
    \end{itemize}

    \begin{figure}
        \centering
        \includegraphics[width=0.8\linewidth]{20240321-ESM/figs/aa2Ab.pdf}
        \caption{The processes from AA sequences to antibody structure and
        a ribbon representation of an intact immunoglobulin (IgG) with PDB ID: \texttt{1igt}\cite{IgG2a1997}}
        \label{fig:aa2Ab}
    \end{figure}
\end{frame}

\note[enumerate]{
    \item First, let's get to know antibody.
    \item We have already known that 20 canonical amino acids can make up a peptide, and then be folded and assembled to a functional protein complex, like antibody.
    \item This is an example from mouse, its PDB ID is 1igt. 
    \item It is consist of two heavy chains and two light chains, and they are connected by the disuphide bridges.
    \item Each chain has variable domain and constant domains.
    \item The variable domains have some small and highly variable regions, named CDR, which accounting for the antibody's specificity to antigen.
    \item Those head of antibody are called Fab and the rest tail is Fc.
}

\begin{frame}{Antibody engineering --- \normalsize Evolution searching}
    To improve the \textbf{fitness of a given antibody} in laboratory,
    \begin{enumerate}\small
        \item Random guessing or brute force searching in \textbf{full mutation space} will cause \textcolor{purple}{considerable experimental burden}.
        % \begin{itemize}
        %     \item devoting substantial effort to weakly active or non-functional proteins
        %     \item requiring high throughput to identify variants with improved fitness
        % \end{itemize}
        \item Narrowing down to a \textbf{evolutionary plausibility space} might \textbf{enrich the high fitness}, but
        % \begin{itemize}
        %     \item The general properties are prerequisites for fitness and function across most proteins.
        %     \item Identifying plausible mutations could help guide evolution away from invalid regimes.
        % \end{itemize}
        % \item \textcolor{purple}{However, the challenges are}
        \begin{enumerate}
            \item protein sequences evolution rule are complex, and
            \item searching space might be still too large to use practically.
        \end{enumerate}
    \end{enumerate}    

    \begin{framed}
        \textbf{Open question}: Whether general evolutionary information is sufficient to enable efficient evolution under specific selection pressures?
    \end{framed}


    \begin{figure}
        \centering
        \begin{columns}
            \column{0.1\linewidth}
            \column{0.5\linewidth}
            \centering
            \includegraphics[width=0.9\linewidth]{20240321-ESM/figs/searchingspaces.pdf}
            \column{0.3\linewidth}
            \caption{Narrow a random-guess space to narrow-down \textbf{plausible spaces} with high evolution possibility in (a) or (b)}
            \label{fig:searchingspaces}
            \column{0.1\linewidth}
        \end{columns}
    \end{figure}

\end{frame}

\note[enumerate]{
    \item To improve the fitness of a given antibody, we need to search across the feasible space of mutated sequence, which may be very large.
    \item In nature, the evolution is based on simply random mutation and recombination under specific selection pressure.
    \item However, the similar approach in laboratory will cause considerable burden due to the tiny ratio of high fitness mutations in full mutational space.
    \item The intuition is to narrow down the space, such as evolutionary plausibility space, because
    % \begin{itemize}
    %     \item certain mutations may cause larger disruptions to secondary or tertiary structures.
    %     \item particular amino acid must be carefully coordinated to carry out the biological role of a protein, or the protein species will become extinct.
    %     \item In general, none of these mutations are entirely restricted, but the bias toward maintenance of the existing fold is sufficiently strong to imprint upon the evolutionary process
    % \end{itemize}
    \item Identifying evolutionary plausible mutations could help guide evolution away from invalid regimes, thereby enrich the high fitness variants.
    \item However, the challenges are that protein sequences evolution rule are complex, and searching space might be still too large to be applied practically.
    \item SO, the \textbf{Open question} is Whether general evolutionary information is sufficient to enable efficient evolution under specific selection pressures?
}{}

\subsection{Pretrained Protein Language Models}

\begin{frame}{Protein Language Models \normalsize --- Transformer Encoder\cite{vaswani_attention_2023}}
    \begin{columns}
        \column{0.05\linewidth}
        \column{0.32\linewidth}
        \begin{figure}
            \centering
            \includegraphics[width=\linewidth]{20240321-ESM/figs/transformer_plm.pdf}
            \caption{Architecture of Transformer Encoder with input sequence and output probabilities}
            \label{fig:transformer_plm}
        \end{figure}
        \column{0.73\linewidth}
        \begin{enumerate}\small
            \item Embedding layer
            \begin{gather}
                \bm{x} \in\texttt{Vocab}\triangleq(\{\texttt{special tokens}\}\cup\mathcal{X})^L \\
                \bm{h}_0 = \texttt{\color{blue}token\_emb}(\bm{x}) + f_\text{pos}(1:L) \in \mathbb{R}^{L\times d}
            \end{gather}
            \item Transformer blocks with $H$-head attention, $n=1:N$
            \begin{align}
                \mathbf{Q}_{n,h} & = \bm{h}_{n-1}{\color{blue}\mathbf{W}_{Q,n,h}} \quad (\mathbf{K}_{n,h},\mathbf{V}_{n,h} \text{ are similar})\\
                \bm{a}_{n,h} & = \underbrace{\texttt{softmax}(\mathbf{Q}_{n,h}\mathbf{K}_{n,h}^\mathsf{T}/\sqrt{d})}_\text{attention score}\mathbf{V}_{n,h} \\
                \bm{a}_n & = \texttt{\color{blue}linear}([\bm{a}_{n,1},\cdots,\bm{a}_{n,H}]) \\
                \Tilde{\bm{h}}_n & = \texttt{Normalize}(\bm{a}_n + \bm{h}_{n-1}) \\
                \bm{h}_n & = \texttt{Normalize}(\texttt{\color{blue}FFN}(\Tilde{\bm{h}}_n) + \Tilde{\bm{h}}_n)
            \end{align}
            \item Output layer
            \begin{gather}
                \hat{\bm{p}}(\bm{X}'\mid\bm{x}) = \texttt{softmax}(\texttt{\color{blue}linear}(\bm{h}_N))\in\mathbb{R}^{L\times|\texttt{Vocab}|}
            \end{gather}
        \end{enumerate}
    \end{columns}
\end{frame}

\note[enumerate]{
    \item The architecture of PLM is a intact Transformer encoder but with protein sequences as inputs and the probabilities of each site as outputs.
    \item Intuitively, it can be viewed as three major parts.
    \item First, the protein sequence is embedded to a hidden variable through embed layer and positional encoding.
    \item Then, it be delivered to many transformer blocks with multi-head attention to capture the information globally. You can see the input and output dimension of transformer block is unchanged.
    \item Last, the final hidden variable are passed to a linear layer and softmax to predict the likelihood of each site and amino acid.
    \item If you need more technical details, please let me know.
}{}{}

\subsection{PLM-guided evolution processes}

\begin{frame}{PLM-guided sequence design \normalsize --- Single AA substitution (Round 1)}
    Given a AA sequence $\bm{x}\in\mathcal{X}^L$, 
    {\footnotesize\begin{align}
        \mathcal{S}_j & =\left\{(i,x_i'):\frac{p_j(x_i'\mid\bm{x})}{p_j(x_i\mid\bm{x})}>{\color{purple}\alpha},1\leq i\leq L,x_i'\in \mathcal{X}\right\}, j=1,\cdots,M \\
        f(i,x'_i) & \triangleq \sum_{j} \mathbb{I}_{\{(i,x'_i)\in\mathcal{M}_j\}} \\
        \mathcal{A} & = \left\{(i,x'_i):f(i,x'_i)\geq {\color{purple}k}, 1\leq i\leq L,x'_i\in\mathcal{X}\right\}
    \end{align}}
    \begin{enumerate}\small
        \item $\mathcal{S}_j$:\textbf{Set of candidate substitutions} preferred by PLM $j$
        \item $f(i,x'_i)$: \textbf{Voting result} for each candidate by the $M$ models
        \item $\mathcal{A}$: \textbf{Set of nominated substitutions} consistent across the models
    \end{enumerate}
    \medskip
    For antibody, there are two sequences ($V_H$ and $V_L$) determining antigen affinity.
    increase ${\color{purple}k}$ (roughly) and ${\color{purple}\alpha}$ (finely) to produce a small number of nominees 
    for each antibody ($|\mathcal{A}_{V_H}\cup \mathcal{A}_{V_L}|\sim 10$).\footnote{feasible for screening via low-throughput assays}
\end{frame}

\note[enumerate]{
    \item After obtaining models, let see the strategy of antibody artificial evolution at single site.
    \item First, we exam the likelihood ratio greater for each site and each amino acid to form set of candidate substitutions preferred by each model
    \item Second, summarize the voting result for each candidate by all models
    \item Finally, set a vote threshold $k$ to obtain the nominated substitutions
    \item For antibody, the above final set includes two chains' substitutions and limits the size to be about 10 by adjusting k and alpha, in favor of subsequent low-thoughput testing.
}{}{}


\begin{frame}{PLM-guided sequence design \normalsize --- Multiple AAs substitutions (Round 2)}
    Those \textbf{variants with enhanced affinity}\footnote{measured by Bio-Layer Immobilization (BLI) binding experiments given its target antigen, mechanism shown in Fig. (\ref{fig:BLI_data}) of Appendix} were recombined
    such that the second-round variants have \textbf{two or more substitutions} from wild-type.
    \begin{itemize}\small
        \item Small amount improved variants from Round 1: all combinations
        \item Large amount improved variants from Round 1: manually selected combinations of the selected set of most improved variants
    \end{itemize}
    \bigskip
    \begin{figure}
        \centering
        \includegraphics[width=0.6\linewidth]{20240321-ESM/figs/round2.pdf}
        \caption{Variant recombination for Round 2 screening}
        \label{fig:round2}
    \end{figure}
\end{frame}

\note[enumerate]{
    \item Those variants with enhanced affinity were recombined to form variants with more substitutions.
    \item For large amount improved variants from Round 1, those with the highest affinity are selected for manually combination.
}{}{}

\begin{frame}{Evaluation metrics \normalsize --- Therapeutic properties by wet lab assays}\label{sec:assay}
    Four important therapeutic properties are considered to evaluate the designed variants:
    \begin{enumerate}
        \item \ulbf{Antigen binding affinity}: dissociation constant $K_d$ (Mol)
        \item \textbf{Thermostability}: melting temperature $T_m$ ($^\text{o}$C)
        \item \textbf{Non-specific Binding}: median fluorescence intensity (MFI) values\footnote{Adhesion to soluble membrane proteins marked with fluorescence beads}
        % \item \textbf{T-cell-mediated immunogenicity}: the number of predicted peptide binders to HLA I/II\footnote{by published softwares, \texttt{NetMHCPan} and \texttt{NetMHCIIPan}}
        \item \textbf{Virus neutralization}: the concentration of antibody that inhibits 50\% (pseudo-)viral infection $\text{IC}_{50}$
    \end{enumerate}
\end{frame}

\note[enumerate]{
    \item Four therapeutic properties are considered.
    \item Antigen binding affinity is indicated by dissociation constant, denoted as $K_d$, the less the better
    \item Thermostability is indicated by melting temperature, denoted as $T_m$, need to above 70 Celsius degree
    \item Non-specific binding is measured by the adhesion to soluble membrane proteins marked with fluorescence beads and indicated by median fluorescence intensity. the lower the better
    \item Virus neutralization is measured by the concentration of antibody that inhibits 50\% viral infection, denoted as $\text{IC}_{50}$

\subsection{Compuational and evolutional efficiency}

\begin{frame}{\normalsize Efficient evolution speeds up the novel antibody discovery}
    With GPU-accelerated hardware\footnote{AMD EPYC Rome 7502P 2.5-GHz CPU and an Nvidia Ampere A40 48GB GPU}
    \begin{enumerate}
        \item per antibody (including both $V_H$ \& $V_L$): $<1\ \text{s}$ on widely available
        \item 742 therapeutically relevant antibodies\footnote{from Thera-SAbDab database}: $\sim 3\ \text{min}$
        \item scale linearly with the number of antibodies
    \end{enumerate}
    \medskip
    \begin{figure}
        \centering
        \includegraphics[width=0.57\linewidth]{20240321-ESM/figs/evolutionefficiency.pdf}
        \caption{Comparison of affinity fold improvements versus experimental scale. 
        The other methods design variants with rounds of cell culture and sorting, which take weeks.}
        \label{fig:evolutionefficiency}
    \end{figure}
\end{frame}

\note[enumerate]{
    \item For computational efficiency, to give predictions for each antibody requires less than one second with GPU.
    \item Even for all 742 current therapeutical antibody, only about three minutes are needed, and the computational complexity are linear with the number of antibodies.
    \item For the evolutional efficiency, in nature mutation, thousands to millions of variant are required, even if it can achieve a higher level maturation.
    \item Recent antibody engineering technologies, Mason and Wellner, reported similar improvement in affinity but require even more variants and time to be tested.
    \item These methods design variants with rounds of cell culture and sorting, which take weeks.
}{}

\section{Discussion}

\begin{frame}{Discussion}
    \begin{enumerate}
        \item General PLMs can guide highly efficient affinity maturation based in the wide-type antibody sequence alone.
        \item The affinity improvements are lower than those from in vivo random evolution, which explores a much larger mutation space, but it costs much more time, labor, and materials.
        \item Evolutionary information alone provides sufficient prior information when selecting small numbers of substitutions to test for improved fitness.
        \item Language models should also serve as a baseline for future machine learning methods using supervision or other task-specific training data.
    \end{enumerate}
\end{frame}
